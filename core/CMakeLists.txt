CMAKE_MINIMUM_REQUIRED(VERSION 2.8.11)
project(botjs)
INCLUDE(../cmake/botjs_config.cmake)

    src/app
    src/block
    src/type
    )

    src/app/BotApp.hpp
    src/app/BotEngine.hpp
    src/app/CoreBlock.hpp

    src/block/LogBuffer.hpp

    src/app/CoreBlock.cpp

    src/block/BotBlock.cpp
    src/type/ShapeData.cpp
    )
